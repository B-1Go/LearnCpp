typedef struct _tagMyst()
{
	int a;
	folat b;
}MYST;



int main()
{
	int i = 0;
	int* pInt;

	pint = &i;

	*pint = 10; // 해당 포인터로 접근해서 10을 널어라 즉, 'i = 10;' 와 같다

	// 구조체와 포인터

	MYSY s = {};

	MYST* pST = &s; // 구조체 포인터 선언

	// 기본자료형 2개가 들어 있는 구조이므로 무엇을 가르키는지 정의 해줘야함

	(*pST).a = 10;
	(*pST).b = 3.14f;

	// 위 식을 축약하면

	pST->a = 10;
	pST->b = 3.14f;

	return 0;
}
